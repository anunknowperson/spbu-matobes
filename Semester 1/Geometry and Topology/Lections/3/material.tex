\begin{definition}
    V - векторное пространство

    ($\cdot, \cdot$) : $V \times V \to \R$

    \begin{enumerate}
        \item (u, u) $\geq 0$
        (u, u) = 0 <=> U = 0

        \item $(u_1 + u_2; v) = (u_1, v_1) + (u_2, v)$
        $(u, v_1 + v_2) = (u_, v_1) + (u_1 v_2)$

        \item $\alpha(u, v) = (\alpha u, v) = (u, \alpha v)$

    \end{enumerate}
